docker run images 				:	docker run -p(port{user port:defined port}) (image name )
list of container running           :docker ps -a 
push to docker hub				:	docker push (image name)

docker delete container 	 		:	docker rm (first 3 letter of the container followed by space)
docker delete by image name	 		:	docker rmi (first 3 letter of the images followed by space)
docker delete by image id			:	docker rmi -f imageid
commit 					 	:	docker commit (container id) (username)/(image id) : tag
